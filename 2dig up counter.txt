#pragma config CPD = OFF        // Data EEPROM Memory Code Protection bit (Data EEPROM code protection off)

#include <xc.h>
#define _XTAL_FREQ 20000000
void main()
{
    int i,j,k;
    TRISB=0x00;
    TRISA=0x00;
 
    
    int a[]={0xc0,0xf9,0xa4,0xb0,0x99,0x92,0x82,0xf8,0x80,0x90};
    while(1)
    {
        for(i=0;i<=9;i++)
        {
           for(j=0;j<=9;j++)
           {
               for(k=0;k<25;k++)
               {
                   PORTA=0x02;
               
               PORTB=a[i];
               __delay_ms(10);
               PORTA=0x01;
               PORTB=a[j];
               __delay_ms(10);
               
                       
               
                       
        }
                
    }
}
 }
